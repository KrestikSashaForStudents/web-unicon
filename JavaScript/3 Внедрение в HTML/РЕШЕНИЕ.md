### 1. Динамическая подгрузка скрипта

**Задача:** подключить внешний JS-файл динамически.

```html
<!DOCTYPE html>
<html lang="ru">
<head>
  <meta charset="UTF-8">
  <title>Dynamic script</title>
</head>
<body>
  <button id="load">Загрузить скрипт</button>

  <script>
    document.getElementById("load").addEventListener("click", () => {
      const script = document.createElement("script")
      script.src = "extra.js"
      document.body.appendChild(script)
    })
  </script>
</body>
</html>
```

Файл `extra.js`:

```javascript
alert("Внешний скрипт подключен динамически!")
```

### 2. Свой мини-`alert`

**Задача:** реализовать кастомное модальное окно.

```html
<!DOCTYPE html>
<html lang="ru">
<head>
  <meta charset="UTF-8">
  <title>Custom alert</title>
  <style>
    .overlay {
      position: fixed;
      inset: 0;
      background: rgba(0,0,0,0.6);
      display: flex;
      align-items: center;
      justify-content: center;
    }
    .modal {
      background: white;
      padding: 20px;
      border-radius: 8px;
      text-align: center;
    }
  </style>
</head>
<body>
  <button onclick="myAlert('Привет из кастомного алерта!')">Показать alert</button>

  <script>
    function myAlert(text) {
      const overlay = document.createElement("div")
      overlay.className = "overlay"

      const modal = document.createElement("div")
      modal.className = "modal"
      modal.innerHTML = `<p>${text}</p><button id="okBtn">OK</button>`

      overlay.appendChild(modal)
      document.body.appendChild(overlay)

      document.getElementById("okBtn").onclick = () => overlay.remove()
    }
  </script>
</body>
</html>
```

### 3. Таймер обратного отсчёта

**Задача:** реализовать обратный отсчёт с валидацией.

```html
<!DOCTYPE html>
<html lang="ru">
<head>
  <meta charset="UTF-8">
  <title>Countdown</title>
</head>
<body>
  <h1 id="timer"></h1>
  <script>
    let seconds = parseInt(prompt("Введите количество секунд:"))
    if (isNaN(seconds) || seconds <= 0) {
      alert("Неверное значение")
    } else {
      const timerEl = document.getElementById("timer")
      timerEl.textContent = seconds

      const id = setInterval(() => {
        seconds--
        timerEl.textContent = seconds
        if (seconds <= 0) {
          clearInterval(id)
          alert("Время вышло!")
        }
      }, 1000)
    }
  </script>
</body>
</html>
```

### 5. Локальное хранилище

**Задача:** сохранить имя и подставлять его при загрузке.

```html
<!DOCTYPE html>
<html lang="ru">
<head>
  <meta charset="UTF-8">
  <title>LocalStorage demo</title>
</head>
<body>
  <h1 id="greeting"></h1>
  <form id="form">
    <input type="text" id="name" placeholder="Введите имя">
    <button type="submit">Сохранить</button>
  </form>
  <button id="forget">Забыть меня</button>

  <script>
    const greeting = document.getElementById("greeting")
    const form = document.getElementById("form")
    const forgetBtn = document.getElementById("forget")

    function updateGreeting() {
      const name = localStorage.getItem("user")
      if (name) {
        greeting.textContent = "Привет, " + name + "! Рад видеть снова"
      } else {
        greeting.textContent = "Гость, добро пожаловать!"
      }
    }

    form.addEventListener("submit", (e) => {
      e.preventDefault()
      const name = document.getElementById("name").value
      localStorage.setItem("user", name)
      updateGreeting()
    })

    forgetBtn.addEventListener("click", () => {
      localStorage.removeItem("user")
      updateGreeting()
    })

    updateGreeting()
  </script>
</body>
</html>
```

### 6. Адаптивный заголовок

**Задача:** изменять заголовок в зависимости от ширины окна.

```html
<!DOCTYPE html>
<html lang="ru">
<head>
  <meta charset="UTF-8">
  <title>Adaptive title</title>
</head>
<body>
  <h1 id="title"></h1>

  <script>
    const title = document.getElementById("title")

    function updateTitle() {
      if (window.innerWidth > 800) {
        title.textContent = "Широкий экран"
        title.style.color = "red"
      } else {
        title.textContent = "Узкий экран"
        title.style.color = "green"
      }
    }

    window.addEventListener("resize", updateTitle)
    updateTitle() 
  </script>
</body>
</html>
```

### 7. Проверка глобальных переменных

**Задача:** проверить, какие переменные попадают в `window`.

```html
<script>
  var a = 1
  let b = 2
  const c = 3

  console.log("a in window:", "a" in window) // true
  console.log("b in window:", "b" in window) // false
  console.log("c in window:", "c" in window) // false

  function checkVar(name) {
    return name in window
  }

  console.log(checkVar("a")) // true
  console.log(checkVar("b")) // false
  console.log(checkVar("c")) // false
</script>
```
